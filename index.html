        function calculateTyG() {

            const trig_mmol = parseFloat(document.getElementById('triglyceride_mmol').value);

            const gluc_mmol = parseFloat(document.getElementById('glucose_mmol').value);


            if (isNaN(trig_mmol) || isNaN(gluc_mmol) || trig_mmol <= 0 || gluc_mmol <= 0) {

                alert("Kérjük, adjon meg érvényes, pozitív számokat mindkét mezőbe!");

                return;

            }


            const trig_mgdl = trig_mmol * 88.57;

            const gluc_mgdl = gluc_mmol * 18.0182;

            const tygIndex = Math.log((trig_mgdl * gluc_mgdl) / 2);

            const finalResult = tygIndex.toFixed(2);


            const resultBox = document.getElementById('result-box');

            const resultValue = document.getElementById('result-value');

            const interpretation = document.getElementById('result-interpretation');


            resultValue.textContent = finalResult;


            if (finalResult < 4.49) {

                interpretation.textContent = "Ez az érték az optimális tartományba esik.";

                resultBox.className = 'optimal';

            } else {

                interpretation.textContent = "Ez az érték inzulinrezisztenciára utalhat. Kérjük, konzultáljon orvosával!";

                resultBox.className = 'risk';

            }

            

            resultBox.style.display = 'block';

        }

    </script>
